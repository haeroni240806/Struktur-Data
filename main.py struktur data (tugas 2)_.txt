# Program validasi nama dan tabel perkalian

nama_benar = "roni"

# Input nama
nama = input("Masukkan nama anda: ").strip().lower()

# Validasi nama
if nama == nama_benar:
    print("Nama benar, lanjut ke program...\n")
    
    try:
        angka = int(input("Masukkan angka (1-10): "))
        
        if 1 <= angka <= 10:
            print(f"\nTabel perkalian {angka}:")
            
            for i in range(1, 11):
                print(f"{angka} x {i} = {angka * i}")
        else:
            print("Angka harus antara 1 sampai 10!")
    
    except ValueError:
        print("Input harus berupa angka!")
